  <span id="sarja"></span>

  <script>
    var R1 = 50, R2 = 10, R3 = 30, Rsarja, Rrinnan;

    function kolmeSarjaan() {

      return Rsarja;
    }

    document.getElementById("sarja").innerHTML =
      "<span style='color: purple;'>R1 = " + R1 + " Ω</span><br>" +
      "<span style='color: green;'>R2 = " + R2 + " Ω</span><br>" +
      "<span style='color: blue;'>R3 = " + R3 + " Ω</span><br>" +
      "<span style='color: red;'> Sarja = " + kolmeSarjaan(R1, R2, R3).toFixed(2) + " Ω</span><br>";

    function kolmeRinnan() {
      var Rrinnan = 1 / ((1 / R1) + (1 / R2) + (1 / R3));
      return Rrinnan;
    }

    document.getElementById("rinnan").innerHTML =
      "<span style='color: MediumSlateBlue;'> Rinnan = " + kolmeRinnan(R1, R2, R3).toFixed(2) + "Ω</span>";

  </script>

Tehtävä 11

Lyhyesti, mitä koodaus tekee. 


  <div id="painikeLaatikko" oncontextmenu="return false;">
    Paina hiiren painikkeita tässä laatikossa.
  </div>

  <script>
    var laatikko = document.getElementById("painikeLaatikko");

    laatikko.addEventListener("mousedown", function (event) {
      event.preventDefault();

      var teksti = "";
      var vari = "";

      if (event.which === 1) {
        teksti = "Vasen painike";
        vari = "yellow";
      } else if (event.which === 2) {
        teksti = "Keskimmäinen painike";
        vari = "Chartreuse";
      } else if (event.which === 3) {
        teksti = "Oikea painike";
        vari = "RGB(18, 213, 213)";
      }

      laatikko.innerHTML = teksti;
      laatikko.style.backgroundColor = vari;
    });
  </script>
Paina hiiren painikkeita tässä laatikossa.

  <div id="vastusLaatikko">
    Paina hiiren painikkeita tässä laatikossa.
  </div>

  <script>
    var vastusLaatikko = document.getElementById("vastusLaatikko");

    vastusLaatikko.addEventListener("mousedown", function (event) {
      event.preventDefault();

      var sarjaVastus = 100;
      var rinnanVastus = 200;
      var teksti = "Se on hiiren painikkeita vasatuksen";

      if (event.altKey && event.button === 0) {
        teksti = "Rinnankytkennän tulos: " + rinnanVastus + " Ω";
      } else if (event.button === 0) {
        teksti = "Sarjakytkennän tulos: " + sarjaVastus + " Ω";
      }

      vastusLaatikko.innerHTML = teksti;
    });
  </script>
Paina hiiren painikkeita tässä laatikossa.
