        document.getElementById("tervehdys").innerHTML = "<span class='vari4 oma'>Hyvää iltapäivää!</span>";

        document.getElementById("tervehdys").innerHTML = "<span class='vari5 oma'>Hyvää iltaa!</span>";
    }

        document.getElementById("tervehdys").innerHTML = "<span class='vari6 oma'>Hyvää yötä!</span>";
    }
  });
  </script>

    document.getElementById("myForm").addEventListener("submit", function (event) {
        event.preventDefault();

        var form = event.target;
        var inputs = form.getElementsByTagName("input");
        var name = form.querySelector('#name').value;
        var email = form.querySelector('#email').value;

        if (name.trim() === "") {
            alert("Nimi puutuu ja ole hyvä anna Nimi.");
            return;
        }
        else if (email.trim() === "") {
            alert("sähköposti puutuu ja ole hyvä anna sähköposti.")
            return;
        }
        for (var i = 0; i < inputs.length; i++) {
            inputs[i].value = "";
            alert("Kiitos yhteydenotostasi ja otamme pikaisesti yhteyttä sinuun.");
        }
        document.getElementById("messageBox").style.display = "block";
        setTimeout(function () {
            document.getElementById("messageBox").style.display = "none";

<button onclick="handleButtonClick()" id="custom-button"> Klikkaa tähän </button>
<h2 id="dateDisplay"></h2>

<script>
function handleButtonClick() {
  var currentDate = new Date();
  var dayOfWeek = currentDate.toLocaleDateString('fi-FI', { weekday: 'long' });
  var month = currentDate.getMonth();

  var suomenkielisetKuukaudet = ["Tammikuu", "Helmikuu", "Maaliskuu", "Huhtikuu", "Toukokuu",
  "Kesäkuu", "Heinäkuu", "Elokuu", "Syyskuu", "Lokakuu", "Marraskuu", "Joulukuu"];

  var kuukausi = suomenkielisetKuukaudet[month];

  var message = kuukausi + " " + dayOfWeek;

  var dateDisplay = document.getElementById("dateDisplay");
  dateDisplay.innerText = message;
  dateDisplay.style.color = "blue";
}
</script>
